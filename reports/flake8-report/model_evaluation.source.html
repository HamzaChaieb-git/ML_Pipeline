114 def create_summary_report(metrics: Dict[str, float], artifact_dir: str) -> None:
115 """Create and save a comprehensive summary report."""
116 report = {
117 "model_performance": {
118 "overall_metrics": {
119 "accuracy": metrics["accuracy"],
120 "roc_auc": metrics["roc_auc"],
121 "log_loss": metrics["log_loss"],
122 },
123 "class_metrics": {
124 "class_0": {
125 "precision": metrics["precision_class_0"],
126 "recall": metrics["recall_class_0"],
127 "f1": metrics["f1_class_0"],
128 },
129 "class_1": {
130 "precision": metrics["precision_class_1"],
131 "recall": metrics["recall_class_1"],
132 "f1": metrics["f1_class_1"],
133 },
134 },
135 "confusion_matrix": {
136 "true_negatives": metrics["true_negatives"],
137 "false_positives": metrics["false_positives"],
138 "false_negatives": metrics["false_negatives"],
139 "true_positives": metrics["true_positives"],
140 },
141 "derived_metrics": {
142 "balanced_accuracy": metrics["balanced_accuracy"],
143 "precision_recall_ratio": metrics["precision_recall_ratio"],
144 "false_positive_rate": metrics["false_positive_rate"],
145 "false_negative_rate": metrics["false_negative_rate"],
146 },
147 },
148 "evaluation_timestamp": datetime.now().isoformat(),
149 "model_recommendation": (
150 "production"
151 if metrics["accuracy"] > 0.85 and metrics["roc_auc"] > 0.85
152 else "staging"
153 ),
154 }
155  
156 with open(os.path.join(artifact_dir, "evaluation_summary.json"), "w") as f:
157 json.dump(report, f, indent=4)
158 mlflow.log_artifact(os.path.join(artifact_dir, "evaluation_summary.json"))

161 def evaluate_model(model: Any, X_test: Any, y_test: Any) -> Dict[str, float]:
162 """
163 Enhanced model evaluation with comprehensive metrics and visualizations.
164  
165 Args:
166 model: Trained model
167 X_test: Testing features
168 y_test: Testing labels
169  
170 Returns:
171 Dictionary containing all computed metrics
172 """
173 # Create artifacts directory under artifacts/evaluation/
174 artifact_dir = os.path.join(
175 "artifacts", "evaluation", datetime.now().strftime("%Y%m%d_%H%M")
176 )
177 os.makedirs(artifact_dir, exist_ok=True)
178  
179 # Get predictions
180 y_pred = model.predict(X_test)
181 y_pred_proba = model.predict_proba(X_test)
182  
183 # Calculate basic metrics
184 metrics = {
185 "accuracy": accuracy_score(y_test, y_pred),
186 "precision": precision_score(y_test, y_pred),
187 "recall": recall_score(y_test, y_pred),
188 "f1": f1_score(y_test, y_pred),
189 "roc_auc": roc_auc_score(y_test, y_pred_proba[:, 1]),
190 "log_loss": log_loss(y_test, y_pred_proba),
191 }
192  
193 # Get detailed classification report
194 report = classification_report(y_test, y_pred, output_dict=True)
195 confusion = confusion_matrix(y_test, y_pred)
196  
197 # Add class-specific metrics
198 metrics.update(
199 {
200 "precision_class_0": report["0"]["precision"],
201 "recall_class_0": report["0"]["recall"],
202 "f1_class_0": report["0"]["f1-score"],
203 "precision_class_1": report["1"]["precision"],
204 "recall_class_1": report["1"]["recall"],
205 "f1_class_1": report["1"]["f1-score"],
206 }
207 )
208  
209 # Add confusion matrix metrics
210 metrics.update(
211 {
212 "true_negatives": int(confusion[0, 0]),
213 "false_positives": int(confusion[0, 1]),
214 "false_negatives": int(confusion[1, 0]),
215 "true_positives": int(confusion[1, 1]),
216 }
217 )
218  
219 # Add advanced metrics
220 metrics.update(
221 {
222 "balanced_accuracy": (metrics["recall_class_0"] + metrics["recall_class_1"])
223 / 2,
224 "precision_recall_ratio": (
225 metrics["precision"] / metrics["recall"] if metrics["recall"] > 0 else 0
226 ),
227 "false_positive_rate": metrics["false_positives"]
228 / (metrics["false_positives"] + metrics["true_negatives"]),
229 "false_negative_rate": metrics["false_negatives"]
230 / (metrics["false_negatives"] + metrics["true_positives"]),
231 "positive_predictive_value": metrics["precision"],
232 "negative_predictive_value": metrics["true_negatives"]
233 / (metrics["true_negatives"] + metrics["false_negatives"]),
234 }
235 )
236  
237 # Log all metrics to MLflow
238 mlflow.log_metrics(metrics)
239  
240 # Create and log evaluation artifacts
241 if isinstance(X_test, pd.DataFrame):
242 feature_names = X_test.columns.tolist()
243 else:
244 feature_names = [f"feature_{i}" for i in range(X_test.shape[1])]
245  
246 create_evaluation_artifacts(y_test, y_pred, y_pred_proba, feature_names)
247  
248 # Create and log summary report
249 create_summary_report(metrics, artifact_dir)
250  
251 # Print evaluation results
252 print("\nDetailed Classification Report:")
253 print(classification_report(y_test, y_pred))
254 print("\nConfusion Matrix:")
255 print(confusion)
256 print("\nKey Metrics:")
257 for name, value in metrics.items():
258 print(f"{name}: {value:.4f}")
259  
260 return metrics
