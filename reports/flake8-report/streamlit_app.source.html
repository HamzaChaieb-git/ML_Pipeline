159 def load_latest_model():
160 models_dir = os.path.join("artifacts", "models")
161 if not os.path.exists(models_dir):
162 st.error("No models found in artifacts/models directory.")
163 return None
164  
165 model_files = [
166 f
167 for f in os.listdir(models_dir)
168 if f.startswith("model_v") and f.endswith(".joblib")
169 ]
170 if not model_files:
171 st.error("No model files found in artifacts/models.")
172 return None
173  
174 latest_model = max(model_files, key=lambda x: x.split("v")[1].split(".joblib")[0])
175 model_path = os.path.join(models_dir, latest_model)
176 try:
177 model = joblib.load(model_path)
178 return model
